.thp(treasure hunter1-2), .tp3(treasure hunter3) - snowy treasure hunter level pack file format WIP
default width and height is 40x30

struct_root {
    u32 sign; // string is "TH" for for treasure hunter 1, "HT" for treasure hunter 2, "T3" for treasure hunter 3
	u32 unk; // value is 16 always
	u32 levelCount;
    
    struct_level {
	    u16 world;
	    u16 mapWidth;
		u16 mapHeight;
		
		u16 tile[x][y];
		u16 object[x][y];
	}
}
